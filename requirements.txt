API + Dashboard Web — interface para visualizar as notícias coletadas.
Acesse http://localhost:8000 após iniciar.
"""
import asyncio
import os
from contextlib import asynccontextmanager
from fastapi import FastAPI, Query, BackgroundTasks
from fastapi.responses import HTMLResponse, JSONResponse
from fastapi.middleware.cors import CORSMiddleware

from database import init_db, get_recent_articles, get_latest_summary, get_collection_logs
from scheduler import run_pipeline, create_scheduler

# ─────────────────────────────────────────────
#  Startup: inicia banco + agendador
# ─────────────────────────────────────────────
scheduler_instance = None

@asynccontextmanager
async def lifespan(app: FastAPI):
    global scheduler_instance
    init_db()
    # Primeira coleta ao iniciar
    asyncio.create_task(run_pipeline())
    # Inicia agendador
    scheduler_instance = create_scheduler()
    scheduler_instance.start()
    yield
    if scheduler_instance:
        scheduler_instance.shutdown()


app = FastAPI(
    title="Saudi Football Monitor",
    description="Monitor de notícias do futebol saudita",
    lifespan=lifespan,
)
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])


# ─────────────────────────────────────────────
#  API Endpoints
# ─────────────────────────────────────────────
@app.get("/api/articles")
async def api_articles(
    hours: int = Query(24, description="Últimas N horas"),
    tier: str = Query(None, description="Filtrar por tier: A, B ou C"),
    limit: int = Query(50),
):
    articles = get_recent_articles(hours=hours, tier=tier, limit=limit)
    return {"count": len(articles), "articles": articles}


@app.get("/api/summary")
async def api_summary():
    summary = get_latest_summary()
    if not summary:
        return {"summary": "Nenhum resumo disponível ainda. Aguarde a primeira coleta."}
    return summary


@app.get("/api/logs")
async def api_logs():
    return {"logs": get_collection_logs(20)}


@app.post("/api/collect")
async def api_collect(background_tasks: BackgroundTasks):
    """Força uma coleta manual imediata."""
    background_tasks.add_task(run_pipeline)
    return {"status": "Coleta iniciada em background"}


@app.get("/api/stats")
async def api_stats():
    articles_24h = get_recent_articles(hours=24, limit=1000)
    tier_counts = {"A": 0, "B": 0, "C": 0}
    lang_counts = {}
    for a in articles_24h:
        tier_counts[a.get("source_tier", "?")] = tier_counts.get(a.get("source_tier", "?"), 0) + 1
        lang = a.get("language", "unknown")
        lang_counts[lang] = lang_counts.get(lang, 0) + 1
    return {
        "total_24h": len(articles_24h),
        "by_tier": tier_counts,
        "by_language": lang_counts,
        "logs": get_collection_logs(5),
    }

<script>
let allArticles = [];
let currentTier = null;

async function loadStats() {
  const r = await fetch('/api/stats');
  const d = await r.json();
  document.getElementById('stat-total').textContent = d.total_24h;
  document.getElementById('stat-a').textContent = d.by_tier.A || 0;
  document.getElementById('stat-b').textContent = d.by_tier.B || 0;
  document.getElementById('stat-c').textContent = d.by_tier.C || 0;
  if (d.logs && d.logs[0]) {
    const t = new Date(d.logs[0].ran_at);
    document.getElementById('stat-last').textContent = t.toLocaleTimeString('pt-BR', {hour:'2-digit',minute:'2-digit'});
  }
  // logs
  const logsEl = document.getElementById('logs-list');
  logsEl.innerHTML = d.logs.map(l => `
    <div class="log-row">
      <span style="color:#94a3b8">${new Date(l.ran_at).toLocaleString('pt-BR')}</span>
      <span class="log-ok">✓ ${l.articles_new} novos</span>
      <span style="color:#64748b">${l.articles_dup} dup</span>
      <span style="color:#64748b">${l.sources_ok} fontes ok</span>
      ${l.sources_fail ? `<span class="log-fail">${l.sources_fail} falhas</span>` : ''}
      ${l.error_msg ? `<span class="log-fail">${l.error_msg.slice(0,60)}</span>` : ''}
    </div>`).join('');
}

async function loadSummary() {
  const r = await fetch('/api/summary');
  const d = await r.json();
  document.getElementById('summary-text').textContent = d.summary_pt || d.summary || '—';
}

async function loadArticles(tier = null) {
  const url = tier ? `/api/articles?limit=100&tier=${tier}` : '/api/articles?limit=100';
  const r = await fetch(url);
  const d = await r.json();
  allArticles = d.articles;
  renderArticles(allArticles);
}

function renderArticles(articles) {
  const el = document.getElementById('article-list');
  if (!articles.length) { el.innerHTML = '<div class="loading">Nenhuma notícia encontrada.</div>'; return; }
  el.innerHTML = articles.map(a => `
    <div class="article">
      <div class="article-header">
        <span class="tier-badge tier-${a.source_tier}">Tier ${a.source_tier}</span>
        <span class="source-name">${a.source_name}</span>
        <span class="lang-badge">${a.language || '?'}</span>
        <span class="source-name" style="margin-left:auto">${a.source_type}</span>
      </div>
      <div class="article-title">
        ${a.url ? `<a href="${a.url}" target="_blank">${a.title_pt || a.title_orig || 'Sem título'}</a>` :
          (a.title_pt || a.title_orig || 'Sem título')}
      </div>
      ${(a.body_pt || a.body_orig) ? `<div class="article-body">${a.body_pt || a.body_orig}</div>` : ''}
      <div class="article-meta">
        ${a.published_at ? new Date(a.published_at).toLocaleString('pt-BR') : ''}
        · score: ${a.relevance_score}
      </div>
    </div>`).join('');
}

function filterTier(tier, btn) {
  currentTier = tier;
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  loadArticles(tier);
}

async function collectNow() {
  showToast('🔄 Coletando...');
  await fetch('/api/collect', { method: 'POST' });
  showToast('✅ Coleta iniciada! Atualizando em 30s...');
  setTimeout(() => { loadStats(); loadArticles(currentTier); loadSummary(); }, 30000);
}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.display = 'block';
  setTimeout(() => t.style.display = 'none', 4000);
}

// Carrega tudo
loadStats(); loadSummary(); loadArticles();
// Auto-refresh a cada 5 minutos
setInterval(() => { loadStats(); loadArticles(currentTier); loadSummary(); }, 300000);
</script>
